# 第 4 章　定址模式

> **本章目標：** 深入理解 x86-64 的六大定址模式，掌握各種記憶體存取方式的語法、使用場景與底層機制，並了解有效位址（Effective Address）的計算方式。

---

## 4.1 什麼是定址模式？

### 4.1.1 定址模式的定義

**定址模式（Addressing Mode）** 是指令指定「操作對象」的方式，決定了 CPU 如何找到運算元（Operand）的實際資料。

對於一條指令 `MOV dst, src`，「src 在哪裡？」可以有很多種答案：
- 直接寫在指令裡（立即值）
- 存在某個暫存器中
- 存在記憶體的某個位址
- 記憶體位址需要動態計算

這些不同的「尋找方式」就是不同的定址模式。

### 4.1.2 為什麼需要多種定址模式？

單一定址模式無法有效率地處理所有程式設計需求：

```c
// 這些操作在組語層面需要不同的定址模式
int x = 5;           // 立即值
int y = x;           // 記憶體讀取（直接定址）
int z = arr[i];      // 計算位址（基底 + 索引 × 比例）
int w = p->field;    // 結構體欄位（基底 + 位移）
```

### 4.1.3 x86-64 定址模式一覽

| 模式 | 運算元寫法 | 資料來源 | 範例 |
|------|-----------|---------|------|
| 立即值 | `imm` | 指令本身 | `MOV EAX, 42` |
| 暫存器 | `reg` | CPU 暫存器 | `MOV EAX, EBX` |
| 直接（絕對）| `[addr]` | 固定記憶體位址 | `MOV EAX, [0x1000]` |
| 暫存器間接 | `[reg]` | 暫存器存放的位址 | `MOV EAX, [RBX]` |
| 基底 + 位移 | `[reg + disp]` | 暫存器 + 常數偏移 | `MOV EAX, [RBP-4]` |
| 基底 + 索引 × 比例 + 位移 | `[base + idx*scale + disp]` | 完整 SIB 計算 | `MOV EAX, [RBX+RCX*4+8]` |

## 4.2 立即值定址（Immediate Addressing）

### 4.2.1 基本概念

立即值（Immediate Value）是**直接編碼在指令的機器碼中**的常數，不需要額外的記憶體存取。

```asm
MOV EAX, 42            ; EAX = 42（十進位）
MOV AL,  0xFF          ; AL  = 255（十六進位）
MOV RBX, 0x1234567890  ; RBX = 長整數（64-bit 立即值）
ADD EAX, 100           ; EAX += 100
AND EAX, 0x0000FFFF    ; 遮罩：保留低 16 位元
```

### 4.2.2 立即值的大小限制

立即值的位元寬度有限制，並非所有指令都接受完整 64-bit 立即值：

```asm
; ✓ MOV 是少數支援完整 64-bit 立即值的指令
MOV RAX, 0xFFFFFFFFFFFFFFFF     ; 合法

; ✗ 大多數指令的立即值最大只到 32-bit（會 sign-extend 到 64-bit）
ADD RAX, 0xFFFFFFFF             ; 合法（sign-extend：0xFFFFFFFF → 0xFFFFFFFFFFFFFFFF = -1）
ADD RAX, 0x100000000            ; 不合法！超過 32-bit

; 解決方案：先用 MOV 載入到暫存器
MOV RCX, 0x100000000
ADD RAX, RCX
```

### 4.2.3 立即值的機器碼表示

以 `MOV EAX, 5` 為例，機器碼為 `B8 05 00 00 00`：

```
B8       05 00 00 00
│        └──────────── 立即值 5（32-bit 小端序）
└─────────────────── Opcode：MOV EAX, imm32
```

立即值越大，指令也越長；x86 有特殊的 8-bit 符號擴展立即值（`imm8`）格式節省空間：

```asm
ADD EAX, 1         ; 機器碼：83 C0 01（3 bytes，使用 imm8）
ADD EAX, 1000      ; 機器碼：05 E8 03 00 00（5 bytes，使用 imm32）
```

## 4.3 暫存器定址（Register Addressing）

### 4.3.1 基本概念

運算元是 CPU 內部的暫存器，不需存取記憶體，是速度最快的定址方式。

```asm
MOV EAX, EBX       ; EAX = EBX（暫存器 → 暫存器）
ADD RAX, RCX       ; RAX = RAX + RCX
XCHG EAX, EBX      ; 交換 EAX 和 EBX 的值
NEG EAX            ; EAX = -EAX（單一運算元）
INC ECX            ; ECX++
```

### 4.3.2 運算元大小必須一致

```asm
MOV EAX, EBX       ; ✓ 兩個都是 32-bit
MOV RAX, RBX       ; ✓ 兩個都是 64-bit
MOV EAX, RBX       ; ✗ 大小不符（32 vs 64）
MOV EAX, BX        ; ✗ 大小不符（32 vs 16）
```

例外：符號擴展（MOVSX）與零擴展（MOVZX）允許不同大小：

```asm
MOVZX EAX, BL      ; EAX = 零擴展的 BL（0 填充高位）
MOVSX EAX, BL      ; EAX = 符號擴展的 BL（符號位填充高位）

; 範例：BL = 0xFF = 255（無號）= -1（有號）
; MOVZX EAX, BL → EAX = 0x000000FF = 255
; MOVSX EAX, BL → EAX = 0xFFFFFFFF = -1（有號擴展）
```

## 4.4 直接定址（Direct / Absolute Addressing）

### 4.4.1 基本概念

以**固定的記憶體位址**存取資料，方括號 `[]` 表示「去這個位址取值」：

```asm
section .data
    count   DD  0
    pi      DD  3

section .text
    MOV EAX, [count]       ; 讀取 count 的值到 EAX
    MOV [count], EBX       ; 將 EBX 的值寫入 count
    ADD EAX, [pi]          ; EAX += pi 的值
```

### 4.4.2 方括號的意義：位址 vs 內容

這是初學者最容易混淆的地方：

```asm
section .data
    x   DD  42              ; 在位址（假設）0x601000 存放值 42

section .text
    MOV EAX, x             ; EAX = 0x601000（x 的位址，非內容）
    MOV EAX, [x]           ; EAX = 42（x 的內容）

    LEA EAX, [x]           ; EAX = 0x601000（取得位址，等同於第一行）
    MOV EAX, [0x601000]    ; EAX = 42（等同於 MOV EAX, [x]）
```

**記憶口訣：** `[...]` 像是 C 語言的 `*`（解參考）；沒有 `[]` 就是位址本身。

```c
// C 語言對應
int  x = 42;
int *p = &x;

*p      ↔  [x] 或 [p]   // 取內容
 p      ↔   x 或 &x     // 取位址
```

### 4.4.3 資料寬度指定

存取記憶體時，有時需要明確指定存取的大小：

```asm
MOV BYTE  [addr], 1     ; 寫入 1 byte
MOV WORD  [addr], 100   ; 寫入 2 bytes
MOV DWORD [addr], 1000  ; 寫入 4 bytes
MOV QWORD [addr], 0     ; 寫入 8 bytes

; 讀取時若目的是暫存器，大小由暫存器決定：
MOV AL,  [addr]         ; 讀 1 byte（AL 是 8-bit）
MOV AX,  [addr]         ; 讀 2 bytes
MOV EAX, [addr]         ; 讀 4 bytes
MOV RAX, [addr]         ; 讀 8 bytes

; 若目的是記憶體，必須明確指定：
ADD DWORD [counter], 1  ; counter += 1（DWORD 是必要的）
INC BYTE  [flag]        ; flag++（BYTE 是必要的）
```

## 4.5 暫存器間接定址（Register Indirect Addressing）

### 4.5.1 基本概念

暫存器中存放的不是資料本身，而是資料所在的**記憶體位址**（即指標）：

```asm
; RBX 中存放位址 0x601000
MOV RBX, 0x601000
MOV EAX, [RBX]         ; EAX = 記憶體[0x601000] 的值
MOV [RBX], ECX         ; 將 ECX 寫入 記憶體[0x601000]
```

### 4.5.2 對應 C 語言的指標操作

```c
int  value = 42;
int *ptr   = &value;

*ptr = 100;            // 透過指標寫入
int x = *ptr;          // 透過指標讀取
```

```asm
; 假設 ptr 存在 RBX 中
MOV DWORD [RBX], 100   ; *ptr = 100
MOV EAX,  [RBX]        ; x = *ptr
```

### 4.5.3 遍歷陣列（搭配指標遞增）

```asm
; 將陣列 arr 的每個元素加 1
; arr 是 4 個 DWORD 的陣列
section .data
    arr DD 10, 20, 30, 40

section .text
    MOV RBX, arr        ; RBX = arr 的起始位址
    MOV ECX, 4          ; 計數器

.loop:
    ADD DWORD [RBX], 1  ; [RBX] += 1
    ADD RBX, 4          ; 移動到下一個 DWORD（4 bytes）
    DEC ECX
    JNZ .loop
```

## 4.6 基底加位移定址（Base + Displacement Addressing）

### 4.6.1 基本概念

在暫存器（基底）的基礎上加上一個**常數偏移量（Displacement）**：

```
有效位址（EA）= 基底暫存器 + 位移量（常數）
```

```asm
MOV EAX, [RBX + 8]     ; 讀取 RBX 位址往後 8 bytes 的值
MOV EAX, [RBP - 4]     ; 讀取 RBP 位址往前 4 bytes 的值
MOV [RSP + 16], RAX    ; 寫入 RSP + 16 的位址
```

### 4.6.2 最常見用途：存取結構體欄位

```c
// C 結構體
struct Point {
    int x;    // 偏移 0
    int y;    // 偏移 4
    int z;    // 偏移 8
};

struct Point p;
p.x = 10;
p.y = 20;
p.z = 30;
```

```asm
; 假設 p 的起始位址存在 RBX 中
%define Point.x  0
%define Point.y  4
%define Point.z  8

MOV DWORD [RBX + Point.x], 10  ; p.x = 10
MOV DWORD [RBX + Point.y], 20  ; p.y = 20
MOV DWORD [RBX + Point.z], 30  ; p.z = 30
```

### 4.6.3 最常見用途：存取堆疊上的區域變數

函式的區域變數存放在堆疊上，以 RBP 為基底存取：

```asm
my_func:
    PUSH RBP
    MOV  RBP, RSP
    SUB  RSP, 16        ; 保留 4 個 int（各 4 bytes）

    ; 區域變數 a, b, c, d 分別在 [RBP-4], [RBP-8], [RBP-12], [RBP-16]
    MOV DWORD [RBP-4],  10  ; a = 10
    MOV DWORD [RBP-8],  20  ; b = 20
    MOV DWORD [RBP-12], 30  ; c = 30
    MOV DWORD [RBP-16], 40  ; d = 40

    MOV EAX, [RBP-4]        ; EAX = a
    ADD EAX, [RBP-8]        ; EAX += b

    MOV RSP, RBP
    POP RBP
    RET
```

### 4.6.4 位移量的範圍

x86 支援兩種位移量大小：

| 位移量大小 | 範圍 | 指令長度影響 |
|-----------|------|------------|
| `disp8`（8-bit 有號）| -128 ～ +127 | 較短（省 3 bytes） |
| `disp32`（32-bit 有號）| -2GB ～ +2GB | 較長 |

組譯器會自動選擇最短的編碼：

```asm
MOV EAX, [RBP - 4]     ; 位移 -4，使用 disp8，指令較短
MOV EAX, [RBP - 200]   ; 位移 -200，超過 -128，使用 disp32
```

## 4.7 基底 + 索引 × 比例 + 位移定址（SIB Addressing）

### 4.7.1 完整的 SIB 公式

SIB（Scale-Index-Base）是 x86 最強大的定址模式，可以在單一指令中計算複雜的位址：

```
有效位址（EA）= Base + Index × Scale + Displacement

  Base        ：基底暫存器（任何通用暫存器）
  Index       ：索引暫存器（除 RSP 外的任何通用暫存器）
  Scale       ：比例因子，只能是 1、2、4 或 8
  Displacement：常數位移（可省略，預設為 0）
```

### 4.7.2 各組合的使用場景

```asm
; 只有 Base（退化為暫存器間接）
MOV EAX, [RBX]

; Base + Displacement（結構體欄位）
MOV EAX, [RBX + 8]

; Base + Index（二維陣列）
MOV EAX, [RBX + RCX]          ; Scale 預設為 1

; Base + Index × Scale（一維陣列，自動計算 byte 偏移）
MOV EAX, [RBX + RCX*4]        ; arr[RCX]，每個元素 4 bytes

; Base + Index × Scale + Displacement（結構體陣列）
MOV EAX, [RBX + RCX*4 + 8]   ; arr[RCX].field_at_offset_8
```

### 4.7.3 一維陣列存取（Index × Scale）

Scale（1/2/4/8）對應不同的資料型別大小：

```asm
section .data
    byte_arr  DB  1, 2, 3, 4, 5      ; 1 byte 元素
    word_arr  DW  10, 20, 30, 40     ; 2 bytes 元素
    int_arr   DD  100, 200, 300      ; 4 bytes 元素
    long_arr  DQ  1000, 2000         ; 8 bytes 元素

section .text
    MOV RBX, 2         ; 索引 i = 2

    ; 存取 byte_arr[2]：Scale = 1
    MOV AL,  [byte_arr + RBX*1]   ; AL = byte_arr[2] = 3

    ; 存取 word_arr[2]：Scale = 2
    MOV AX,  [word_arr + RBX*2]   ; AX = word_arr[2] = 30

    ; 存取 int_arr[2]：Scale = 4
    MOV EAX, [int_arr + RBX*4]    ; EAX = int_arr[2] = 300

    ; 存取 long_arr[1]：Scale = 8（i = 1）
    MOV RBX, 1
    MOV RAX, [long_arr + RBX*8]   ; RAX = long_arr[1] = 2000
```

### 4.7.4 結構體陣列（SIB + Displacement）

```c
// C 語言
struct Employee {
    int id;       // 偏移 0，4 bytes
    int age;      // 偏移 4，4 bytes
    int salary;   // 偏移 8，4 bytes
};
// sizeof(Employee) = 12，但為了對齊通常為 12

struct Employee team[10];
// 取得 team[i].salary
int s = team[i].salary;
```

```asm
; 假設 team 的位址在 RBX，索引 i 在 RCX
; sizeof(Employee) = 12（無法用 scale，需要另行計算或用乘法）
; 若 sizeof = 8（2 個 int + padding），則可用 scale=8

; 若結構體大小恰好是 4、8 等 2 的冪次：
%define SIZEOF_EMP 8    ; 假設 8 bytes（兩個 int，id 和 age）
%define EMP_salary 4    ; salary 在偏移 4

MOV EAX, [RBX + RCX*8 + EMP_salary]   ; team[i].salary
```

### 4.7.5 Scale 的限制

```asm
; Scale 只能是 1、2、4、8（對應 byte/word/dword/qword）
MOV EAX, [RBX + RCX*4]    ; ✓ Scale = 4
MOV EAX, [RBX + RCX*3]    ; ✗ Scale = 3，不合法！
MOV EAX, [RBX + RCX*6]    ; ✗ Scale = 6，不合法！

; 若需要 stride = 3，只能用乘法指令先計算位址：
MOV EAX, RCX
IMUL EAX, 3
MOV EAX, [RBX + RAX]
```

## 4.8 有效位址（Effective Address）與 LEA 指令

### 4.8.1 有效位址的概念

**有效位址（Effective Address, EA）** 是 CPU 依照定址模式計算出來的最終記憶體位址，計算公式即 SIB 公式。

對於 `MOV EAX, [RBX + RCX*4 + 8]`：
1. 計算有效位址：EA = RBX + RCX×4 + 8
2. 以 EA 為位址讀取 4 bytes
3. 結果存入 EAX

### 4.8.2 LEA 指令：載入有效位址

`LEA`（Load Effective Address）**只計算有效位址，不存取記憶體**，將計算結果存入目的暫存器：

```asm
; MOV 會存取記憶體；LEA 只計算位址
MOV EAX, [RBX + 8]     ; EAX = 記憶體[RBX+8] 的內容
LEA EAX, [RBX + 8]     ; EAX = RBX + 8（位址本身，不存取記憶體）
```

### 4.8.3 LEA 的常見用途

#### 用途 1：取得變數位址（類似 C 的 `&`）

```asm
section .data
    buffer RESB 64

section .text
    LEA RDI, [buffer]   ; RDI = buffer 的位址（等同於 &buffer）
    ; 等同於：MOV RDI, buffer（但 LEA 更通用）
```

#### 用途 2：快速整數乘法（利用 Scale 和 Displacement）

```asm
; EAX = EBX × 5
LEA EAX, [EBX + EBX*4]     ; EAX = EBX + EBX×4 = EBX×5

; EAX = EBX × 3
LEA EAX, [EBX + EBX*2]     ; EAX = EBX + EBX×2 = EBX×3

; EAX = EBX × 9
LEA EAX, [EBX + EBX*8]     ; EAX = EBX + EBX×8 = EBX×9

; EAX = EBX × 10
LEA EAX, [EBX*2 + EBX*8]   ; ✗ 不合法（Base 和 Index 都需要是暫存器）
LEA EAX, [EBX + EBX*4]
ADD EAX, EAX               ; EAX = (EBX×5) × 2 = EBX×10  ✓
```

#### 用途 3：同時做加法與加法（省指令）

```asm
; 不用 LEA：3 條指令
MOV EAX, EBX
ADD EAX, ECX
ADD EAX, 100

; 用 LEA：1 條指令
LEA EAX, [EBX + ECX + 100]
```

#### 用途 4：RIP 相對位址（x86-64 常見）

```asm
; 取得 msg 的絕對位址（用於 PIC，位置無關程式碼）
LEA RDI, [RIP + msg]   ; RDI = msg 的執行時位址
```

## 4.9 x86-64 的定址限制與注意事項

### 4.9.1 不能兩個運算元都是記憶體

```asm
MOV [dst], [src]    ; ✗ 不合法！兩個都是記憶體位址

; 正確做法：透過暫存器中轉
MOV EAX, [src]      ; 先讀到暫存器
MOV [dst], EAX      ; 再寫入目的
```

### 4.9.2 SIB 的 Index 不能是 RSP

```asm
MOV EAX, [RBX + RSP*4]    ; ✗ Index 不能是 RSP
MOV EAX, [RSP + RBX*4]    ; ✓ RSP 可以是 Base（但 Index 不行）
```

原因：x86 編碼中，RSP 在 SIB 位元組的 Index 欄位有特殊含義（表示「沒有索引」）。

### 4.9.3 位址必須在有效的記憶體範圍內

```asm
MOV EAX, [0]              ; ✗ 存取 NULL 位址，會觸發 Segmentation Fault
MOV EAX, [0xFFFF000000]   ; ✗ 超出使用者空間，觸發 Segmentation Fault
```

### 4.9.4 記憶體對齊（Alignment）

雖然 x86 允許非對齊的記憶體存取，但效能較差：

```
最佳存取對齊規則：
  BYTE  （1 byte）：無限制
  WORD  （2 bytes）：位址應為 2 的倍數（位址 & 1 == 0）
  DWORD （4 bytes）：位址應為 4 的倍數（位址 & 3 == 0）
  QWORD （8 bytes）：位址應為 8 的倍數（位址 & 7 == 0）
  XMM   （16 bytes）：位址應為 16 的倍數（MOVAPS 指令強制要求）
```

```asm
; 若資料未對齊，存取仍可能成功但速度較慢
; SSE 指令中，movaps（aligned）比 movups（unaligned）快
MOVAPS XMM0, [aligned_data]     ; 要求 16-byte 對齊，否則 crash
MOVUPS XMM0, [unaligned_data]   ; 允許非對齊，但略慢
```

## 4.10 定址模式的機器碼編碼

### 4.10.1 ModRM 位元組

x86 指令使用 **ModRM** 位元組來編碼定址模式，緊接在 Opcode 之後：

```
ModRM 位元組（8-bit）：
  Bit 7～6：Mod（2 bits）  → 定址模式
  Bit 5～3：Reg（3 bits）  → 目的暫存器 或 Opcode 擴展
  Bit 2～0：R/M（3 bits）  → 來源暫存器 或 記憶體定址

Mod 欄位含義：
  00 → 記憶體，無位移（[reg]）
  01 → 記憶體，8-bit 位移（[reg + disp8]）
  10 → 記憶體，32-bit 位移（[reg + disp32]）
  11 → 暫存器直接（reg）
```

### 4.10.2 SIB 位元組

當 ModRM 的 R/M 欄位為 `100`（RSP）時，代表後面跟著一個 **SIB** 位元組：

```
SIB 位元組（8-bit）：
  Bit 7～6：Scale（2 bits） → 00=×1, 01=×2, 10=×4, 11=×8
  Bit 5～3：Index（3 bits） → 索引暫存器編號
  Bit 2～0：Base（3 bits）  → 基底暫存器編號
```

### 4.10.3 指令編碼範例

```
MOV EAX, [RBX]
  Opcode：8B（MOV r32, r/m32）
  ModRM ：03（Mod=00, Reg=000（EAX）, R/M=011（RBX））
  機器碼：8B 03

MOV EAX, [RBX + 8]
  Opcode：8B
  ModRM ：43（Mod=01, Reg=000（EAX）, R/M=011（RBX））
  Disp8 ：08
  機器碼：8B 43 08

MOV EAX, [RBX + RCX*4]
  Opcode：8B
  ModRM ：04（Mod=00, Reg=000（EAX）, R/M=100（SIB follows））
  SIB   ：8B（Scale=10（×4）, Index=001（RCX）, Base=011（RBX））
  機器碼：8B 04 8B
```

## 4.11 定址模式的實際應用

### 4.11.1 字串複製（MOVSB）

```asm
; 複製 n 個 bytes 從 src 到 dst
; 等同於 C 的 memcpy(dst, src, n)

section .text
    ; 參數（System V ABI）：
    ; RDI = dst，RSI = src，RDX = n
memcpy_impl:
    MOV RCX, RDX       ; RCX = n（計數器）
    CLD                ; 清除 DF，方向為正向（遞增）
    REP MOVSB          ; 重複：[RDI] = [RSI]，RDI++，RSI++，RCX--
    RET
```

### 4.11.2 找陣列最大值

```asm
; 找 int 陣列的最大值
; 輸入：RDI = 陣列位址，RSI = 元素個數
; 輸出：EAX = 最大值

find_max:
    MOV EAX, [RDI]         ; 初始最大值 = 第一個元素
    MOV RCX, 1             ; 從第二個元素開始（index = 1）

.loop:
    CMP RCX, RSI           ; i < n?
    JGE .done

    MOV EDX, [RDI + RCX*4] ; EDX = arr[i]（SIB 定址）
    CMP EDX, EAX
    JLE .skip
    MOV EAX, EDX           ; 更新最大值
.skip:
    INC RCX
    JMP .loop

.done:
    RET
```

### 4.11.3 二維陣列存取

```c
// C 語言：int matrix[M][N]
// 存取 matrix[row][col]
// 位址 = matrix + (row × N + col) × sizeof(int)
```

```asm
; 假設：matrix 位址在 RBX，N（每列元素數）= 4，int = 4 bytes
; 要存取 matrix[row][col]，row 在 RCX，col 在 RDX

%define N 4             ; 每列元素數

    ; 計算位址：row × N + col
    MOV EAX, ECX       ; EAX = row
    IMUL EAX, N        ; EAX = row × N
    ADD EAX, EDX       ; EAX = row × N + col（線性 index）

    ; 存取元素（Scale = 4，每個 int 4 bytes）
    MOV ECX, EAX
    MOV EAX, [RBX + RCX*4]    ; EAX = matrix[row][col]
```

## 4.12 本章小結

**六大定址模式回顧：**

| 模式 | 語法 | 有效位址計算 | 典型用途 |
|------|------|------------|---------|
| 立即值 | `imm` | 指令內含 | 常數、初始值 |
| 暫存器 | `reg` | 無需記憶體 | 臨時值、迴圈變數 |
| 直接 | `[addr]` | addr | 全域變數存取 |
| 暫存器間接 | `[reg]` | reg | 指標解參考 |
| 基底+位移 | `[reg+disp]` | reg + disp | 結構體欄位、堆疊區域變數 |
| SIB | `[base+idx*scale+disp]` | base + idx×scale + disp | 陣列、結構體陣列 |

**LEA vs MOV：**
- `MOV EAX, [expr]`：計算位址，存取記憶體，EAX = 記憶體內容
- `LEA EAX, [expr]`：計算位址，**不存取記憶體**，EAX = 位址本身

**重要限制：**
- 兩個運算元不能同時是記憶體位址
- SIB 的 Index 不能是 RSP
- Scale 只能是 1、2、4、8

**ModRM/SIB 編碼：**
- 定址模式被編碼在 ModRM 位元組的 Mod 欄位
- 複雜的 SIB 模式需要額外的 SIB 位元組
- 組譯器自動選擇最短的 disp8 或 disp32 編碼
