# 第 7 章　巨集處理器

> **本章目標：** 深入理解巨集處理器的運作原理、巨集定義與展開的機制、各種進階巨集技巧，以及巨集在現代組譯器（NASM）和 C 前置處理器中的實際應用與差異。

---

## 7.1 巨集的基本概念

### 7.1.1 什麼是巨集？

**巨集（Macro）** 是一段有名稱的程式碼模板，在組譯或編譯前由巨集處理器（Macro Processor）自動展開為對應的程式碼。使用巨集的目的是：

- **消除重複程式碼**：相同的操作序列只需定義一次，多處使用
- **提高可讀性**：用有意義的名稱取代一長串指令
- **參數化**：同一個模板可以用不同參數產生不同的程式碼
- **條件控制**：依據條件選擇性地產生不同程式碼

### 7.1.2 巨集 vs 副程式（函式）

巨集和副程式都能消除重複程式碼，但運作方式根本不同：

| 比較項目 | 巨集 | 副程式（函式）|
|---------|------|-------------|
| 展開時機 | **組譯前**（文字替換）| 執行時（CALL 跳入）|
| 執行方式 | 程式碼直接嵌入（每次展開一份）| 共用同一份程式碼 |
| 呼叫開銷 | **無**（無 CALL/RET）| 有（PUSH/POP、堆疊操作）|
| 程式碼大小 | **較大**（每次使用都複製一份）| 較小（只有一份）|
| 參數傳遞 | 文字替換（組譯時期）| 暫存器或堆疊（執行時期）|
| 遞迴 | **不支援** | 支援 |
| 區域標號衝突 | **需要特殊處理** | 不需要（各自的作用域）|
| 適用場景 | 短小、頻繁呼叫的程式碼 | 較長、複雜的程式碼 |

```asm
; 副程式版本：程式碼只有一份，每次 CALL
print_newline_sub:
    MOV RAX, 1
    MOV RDI, 1
    LEA RSI, [newline]
    MOV RDX, 1
    SYSCALL
    RET

; 使用時：
    CALL print_newline_sub   ; 有 CALL/RET 開銷
    CALL print_newline_sub

; 巨集版本：每次展開都複製一份程式碼
%macro PRINT_NEWLINE 0
    MOV RAX, 1
    MOV RDI, 1
    LEA RSI, [newline]
    MOV RDX, 1
    SYSCALL
%endmacro

; 使用時：
    PRINT_NEWLINE            ; 直接展開，無呼叫開銷
    PRINT_NEWLINE            ; 再展開一份（共兩份程式碼）
```

### 7.1.3 巨集處理器的處理流程

```
原始組語程式碼（含巨集定義與呼叫）
            │
            ▼
    ┌──────────────────┐
    │   巨集處理器      │
    │  1. 掃描巨集定義  │
    │  2. 建立巨集定義表│
    │  3. 展開巨集呼叫  │
    └──────────────────┘
            │
            ▼
    展開後的純組合語言
    （不含任何巨集定義/呼叫）
            │
            ▼
    ┌──────────────────┐
    │  兩遍組譯器       │
    │  Pass 1 / Pass 2 │
    └──────────────────┘
            │
            ▼
         目的碼
```

## 7.2 NASM 巨集語法

### 7.2.1 基本巨集定義

```asm
%macro 巨集名稱 參數個數
    ; 巨集主體（可包含任何合法的組語）
%endmacro
```

#### 無參數巨集

```asm
; 定義：印出換行
%macro NEWLINE 0
    MOV  RAX, 1          ; sys_write
    MOV  RDI, 1          ; stdout
    LEA  RSI, [nl_char]  ; 換行字元的位址
    MOV  RDX, 1          ; 長度
    SYSCALL
%endmacro

; 使用：
    NEWLINE              ; 展開為上述 5 條指令
```

#### 帶參數巨集（`%1`、`%2`...）

```asm
; 定義：印出任意字串
; %1 = 字串標號，%2 = 字串長度
%macro PRINT_STR 2
    MOV  RAX, 1
    MOV  RDI, 1
    LEA  RSI, [%1]       ; %1 被替換為第一個引數
    MOV  RDX, %2         ; %2 被替換為第二個引數
    SYSCALL
%endmacro

section .data
    hello     DB "Hello, World!", 0x0a
    hello_len EQU $ - hello

section .text
    PRINT_STR hello, hello_len   ; 展開時 %1→hello，%2→hello_len
```

**展開後等同於：**

```asm
    MOV  RAX, 1
    MOV  RDI, 1
    LEA  RSI, [hello]
    MOV  RDX, hello_len
    SYSCALL
```

### 7.2.2 巨集展開的本質：文字替換

巨集展開是**純粹的文字替換**，參數不經過任何型別判斷：

```asm
%macro ADD_TO 2
    ADD %1, %2
%endmacro

    ADD_TO EAX, EBX      ; → ADD EAX, EBX
    ADD_TO [var], 10     ; → ADD [var], 10
    ADD_TO EAX, [arr+4]  ; → ADD EAX, [arr+4]
```

因為是純文字替換，傳入任何「文字上合法」的內容都能被接受（至於是否產生正確的組語，由組譯器判斷）：

```asm
    ADD_TO EAX, EAX      ; → ADD EAX, EAX  （合法，等同 SHL EAX,1）
    ADD_TO [m1], [m2]    ; → ADD [m1], [m2] （不合法！兩個都是記憶體）
```

### 7.2.3 引數個數的彈性：`%0` 與預設值

```asm
; %0 代表傳入的引數總數
%macro SHOW_ARGS 0-3   ; 接受 0 到 3 個引數
    %if %0 >= 1
        MOV EAX, %1
    %endif
    %if %0 >= 2
        MOV EBX, %2
    %endif
    %if %0 >= 3
        MOV ECX, %3
    %endif
%endmacro

    SHOW_ARGS 10          ; %0=1，只設 EAX
    SHOW_ARGS 10, 20      ; %0=2，設 EAX 和 EBX
    SHOW_ARGS 10, 20, 30  ; %0=3，設 EAX、EBX、ECX
```

## 7.3 巨集中的標號問題與解決

### 7.3.1 標號衝突問題

巨集展開時，若巨集內含有標號，多次展開會導致**重複標號定義**錯誤：

```asm
%macro CHECK_ZERO 1
    TEST %1, %1
    JZ   .is_zero        ; ← 問題：每次展開都定義 .is_zero
    MOV  EBX, 1
    JMP  .done
.is_zero:
    MOV  EBX, 0
.done:
%endmacro

    CHECK_ZERO EAX       ; 展開 → 定義 .is_zero 和 .done
    CHECK_ZERO EBX       ; ❌ 再次展開 → 重複定義！
```

### 7.3.2 解決方案：`%%label` 自動產生唯一標號

NASM 的 `%%label` 語法會為每次展開自動產生唯一的標號名稱：

```asm
%macro CHECK_ZERO 1
    TEST %1, %1
    JZ   %%is_zero       ; NASM 自動替換為唯一名稱
    MOV  EBX, 1
    JMP  %%done
%%is_zero:
    MOV  EBX, 0
%%done:
%endmacro

; 第一次展開：%%is_zero → ..@1.is_zero，%%done → ..@1.done
    CHECK_ZERO EAX

; 第二次展開：%%is_zero → ..@2.is_zero，%%done → ..@2.done
    CHECK_ZERO EBX       ; ✓ 標號名稱不同，不衝突
```

## 7.4 進階巨集技巧

### 7.4.1 巨集內呼叫其他巨集（巢狀巨集）

```asm
%macro SAVE_REGS 0
    PUSH RAX
    PUSH RBX
    PUSH RCX
%endmacro

%macro RESTORE_REGS 0
    POP  RCX
    POP  RBX
    POP  RAX
%endmacro

%macro WITH_SAVED_REGS 0-*   ; 接受可變引數
    SAVE_REGS                ; 呼叫另一個巨集
    %rotate 0                ; 不做旋轉，只是示意可以有多行主體
    RESTORE_REGS
%endmacro
```

### 7.4.2 `%rep` — 重複展開

```asm
; 展開 10 次 NOP
%rep 10
    NOP
%endrep

; 產生查找表
%assign i 0
%rep 256
    DB i * 2          ; 產生 0, 2, 4, ..., 510 的查找表
    %assign i i + 1
%endrep
```

### 7.4.3 `%rotate` — 引數旋轉

```asm
; 將多個引數逐一處理
%macro PUSH_ALL 1-*      ; 1 個以上的可變引數
    %rep %0              ; 重複 %0（引數個數）次
        PUSH %1          ; 每次 PUSH 第一個引數
        %rotate 1        ; 將所有引數向左旋轉一位（%2 變成新的 %1）
    %endrep
%endmacro

%macro POP_ALL 1-*
    %rep %0
        %rotate -1       ; 向右旋轉（反向還原）
        POP %1
    %endrep
%endmacro

; 使用：
    PUSH_ALL RAX, RBX, RCX    ; → PUSH RAX; PUSH RBX; PUSH RCX
    POP_ALL  RAX, RBX, RCX    ; → POP RCX; POP RBX; POP RAX（反向）
```

### 7.4.4 `%assign` 與 `%define` 的差異

```asm
; %define：文字替換（類似 C 的 #define）
%define BUFFER_SIZE 1024
%define MAX(a, b)  ( (a) > (b) ? (a) : (b) )   ; 巨集函式（組語不支援，但 NASM 可用於常數計算）

buf    RESB BUFFER_SIZE    ; → RESB 1024

; %assign：數值常數（只接受整數，在組譯時期求值）
%assign COUNTER 0
%assign COUNTER COUNTER + 1   ; 累加（%define 做不到這個）
%assign COUNTER COUNTER + 1

; 此時 COUNTER = 2
    MOV EAX, COUNTER           ; → MOV EAX, 2
```

### 7.4.5 字串化引數（`%[...]` 和 `%!`）

```asm
; 將引數轉為字串（用於資料定義）
%macro DEFINE_STRING 2
    %1   DB  %2, 0       ; %1 是標號，%2 是字串內容
%endmacro

    DEFINE_STRING greeting, "Hello!"
    ; 展開為：greeting DB "Hello!", 0
```

## 7.5 條件組譯

### 7.5.1 `%ifdef` / `%ifndef`

```asm
%define DEBUG_BUILD    ; 定義 DEBUG_BUILD 符號

%ifdef DEBUG_BUILD
    ; 只在 DEBUG 模式下組譯這段
    MOV EAX, 0xDEBD   ; 除錯用的哨兵值
    CALL print_debug
%endif

%ifndef RELEASE
    %warning "注意：非 Release 模式"
%endif
```

### 7.5.2 `%if` / `%elif` / `%else` / `%endif`

```asm
%define TARGET_ARCH 64      ; 目標架構位元數

%if TARGET_ARCH == 64
    %define PTR_SIZE 8
    %define MOV_PTR  MOV QWORD
%elif TARGET_ARCH == 32
    %define PTR_SIZE 4
    %define MOV_PTR  MOV DWORD
%else
    %error "不支援的架構"
%endif

    ; PTR_SIZE 和 MOV_PTR 已在組譯時決定
    RESB PTR_SIZE       ; 保留一個指標大小的空間
```

### 7.5.3 `%error` 與 `%warning`

```asm
%if BUFFER_SIZE < 64
    %error "BUFFER_SIZE 必須至少 64 bytes！"
%endif

%if BUFFER_SIZE > 65536
    %warning "BUFFER_SIZE 很大，可能影響效能"
%endif
```

## 7.7 C 前置處理器（C Preprocessor）

C 語言本身也有一套巨集系統，在編譯前由前置處理器（cpp）處理。

### 7.7.1 `#define` — 物件式巨集

```c
#define PI          3.14159265358979
#define MAX_SIZE    1024
#define TRUE        1
#define FALSE       0
#define NULL        ((void*)0)

// 使用：
double area = PI * r * r;
int buf[MAX_SIZE];
```

### 7.7.2 `#define` — 函式式巨集

```c
// 帶參數的巨集（注意：每個參數和整體都要加括號！）
#define MAX(a, b)          ((a) > (b) ? (a) : (b))
#define MIN(a, b)          ((a) < (b) ? (a) : (b))
#define ABS(x)             ((x) < 0 ? -(x) : (x))
#define SQUARE(x)          ((x) * (x))
#define ARRAY_SIZE(arr)    (sizeof(arr) / sizeof((arr)[0]))
#define OFFSETOF(type, member)  ((size_t)&((type*)0)->member)

// 使用：
int bigger = MAX(a, b);
int arr[10];
int n = ARRAY_SIZE(arr);   // = 10
```

**為什麼要加括號？**

```c
// 危險的巨集（未加括號）：
#define BAD_SQUARE(x)  x * x
// BAD_SQUARE(1 + 2) → 1 + 2 * 1 + 2 = 5（錯！應為 9）

// 正確的巨集：
#define GOOD_SQUARE(x) ((x) * (x))
// GOOD_SQUARE(1 + 2) → ((1+2) * (1+2)) = 9（正確）
```

### 7.7.3 多行巨集（反斜線連行）

```c
#define SWAP(type, a, b) do {   \
    type _tmp = (a);             \
    (a) = (b);                   \
    (b) = _tmp;                  \
} while (0)

// 使用：
SWAP(int, x, y);

// 展開後：
do {
    int _tmp = (x);
    (x) = (y);
    (y) = _tmp;
} while (0);
```

**為什麼用 `do { ... } while(0)`？**

```c
// 若直接用 { }，在 if 中使用時會出問題：
#define BAD_SWAP(a, b) { int t=a; a=b; b=t; }
if (cond)
    BAD_SWAP(x, y);   // → if (cond) { int t=x; ... }; ← 多餘的分號
else
    ...               // ❌ 語法錯誤！

// do-while(0) 讓巨集整體成為一個「語句」
#define GOOD_SWAP(a, b) do { int t=a; a=b; b=t; } while(0)
if (cond)
    GOOD_SWAP(x, y);  // → if (cond) do { ... } while(0); ← 正確
else
    ...
```

### 7.7.4 預定義巨集

C 前置處理器預先定義了一些有用的巨集：

| 巨集 | 說明 | 範例值 |
|------|------|--------|
| `__FILE__` | 目前原始檔名（字串）| `"main.c"` |
| `__LINE__` | 目前行號（整數）| `42` |
| `__func__` | 目前函式名稱（C99）| `"main"` |
| `__DATE__` | 編譯日期（字串）| `"Jan 01 2025"` |
| `__TIME__` | 編譯時間（字串）| `"12:00:00"` |
| `__STDC__` | 符合 ANSI C 標準 | `1` |
| `__cplusplus` | C++ 編譯器 | `202002L` |

```c
// 除錯輸出（附帶位置資訊）
#define DEBUG_PRINT(fmt, ...) \
    printf("[%s:%d] " fmt "\n", __FILE__, __LINE__, ##__VA_ARGS__)

DEBUG_PRINT("x = %d", x);
// 展開：printf("[main.c:42] x = %d\n", x);
```

### 7.7.5 `#ifdef` / `#ifndef` — 條件編譯

```c
// 防止標頭檔重複包含（Include Guard）
#ifndef MYLIB_H
#define MYLIB_H

// 標頭檔內容...
void my_function(int x);

#endif   // MYLIB_H

// 或使用較新的非標準寫法：
#pragma once    // 大多數編譯器支援，效果相同
```

```c
// 跨平台條件編譯
#ifdef _WIN32
    #include <windows.h>
    #define SLEEP(ms) Sleep(ms)
#elif defined(__linux__)
    #include <unistd.h>
    #define SLEEP(ms) usleep((ms) * 1000)
#elif defined(__APPLE__)
    #include <unistd.h>
    #define SLEEP(ms) usleep((ms) * 1000)
#endif
```

### 7.7.6 可變引數巨集（`__VA_ARGS__`）

```c
// C99 以後支援
#define LOG(level, fmt, ...) \
    fprintf(stderr, "[%s] " fmt "\n", level, ##__VA_ARGS__)

LOG("INFO",  "程式啟動");
LOG("ERROR", "開啟檔案失敗：%s", filename);
```

## 7.8 巨集與副程式的混合使用策略

### 7.8.1 什麼情況下用巨集，什麼情況下用副程式？

```
用巨集（Macro）：
  ✓ 程式碼極短（≤ 5 條指令）
  ✓ 呼叫頻率極高（迴圈中、熱路徑）
  ✓ 需要在組譯時期決定的行為
  ✓ 包裝固定的程式碼模式（如函式序言/結語）
  ✓ 跨平台的條件程式碼

用副程式（Subroutine）：
  ✓ 程式碼較長（> 10 條指令）
  ✓ 需要遞迴
  ✓ 程式碼大小比執行速度更重要
  ✓ 需要明確的作用域與除錯支援
```

### 7.8.2 實際應用：系統呼叫包裝巨集

```asm
; Linux x86-64 系統呼叫包裝
%macro SYSCALL1 2          ; syscall_num, arg1
    MOV RAX, %1
    MOV RDI, %2
    SYSCALL
%endmacro

%macro SYSCALL3 4          ; syscall_num, arg1, arg2, arg3
    MOV RAX, %1
    MOV RDI, %2
    MOV RSI, %3
    MOV RDX, %4
    SYSCALL
%endmacro

; 常用系統呼叫編號
%define SYS_READ    0
%define SYS_WRITE   1
%define SYS_EXIT    60

; 使用：
    SYSCALL3 SYS_WRITE, 1, msg, msg_len    ; write(1, msg, msg_len)
    SYSCALL1 SYS_EXIT,  0                  ; exit(0)
```

### 7.8.3 實際應用：函式框架巨集

```asm
; 自動產生函式序言與結語
%macro FUNC_BEGIN 1        ; 引數：需要保留的堆疊空間（bytes）
    PUSH RBP
    MOV  RBP, RSP
    %if %1 > 0
        SUB  RSP, %1
    %endif
%endmacro

%macro FUNC_END 0
    MOV  RSP, RBP
    POP  RBP
    RET
%endmacro

; 使用：
my_func:
    FUNC_BEGIN 32          ; 展開為 PUSH RBP; MOV RBP,RSP; SUB RSP,32
    ; ... 函式主體 ...
    FUNC_END               ; 展開為 MOV RSP,RBP; POP RBP; RET
```

## 7.9 巨集定義表（Macro Definition Table）

### 7.9.1 巨集處理器的內部資料結構

巨集處理器在第一遍掃描時，將所有巨集定義存入**巨集定義表（MDT，Macro Definition Table）**：

```
巨集定義表（MDT）
┌──────────────┬──────────┬─────────────────────────────────┐
│  巨集名稱    │ 引數個數 │  巨集主體（替換文字）            │
├──────────────┼──────────┼─────────────────────────────────┤
│  PRINT_STR   │    2     │  MOV RAX,1                      │
│              │          │  MOV RDI,1                      │
│              │          │  LEA RSI,[?1]                   │  ← ?1 代表 %1
│              │          │  MOV RDX,?2                     │  ← ?2 代表 %2
│              │          │  SYSCALL                        │
├──────────────┼──────────┼─────────────────────────────────┤
│  NEWLINE     │    0     │  MOV RAX,1                      │
│              │          │  MOV RDI,1                      │
│              │          │  ...                            │
└──────────────┴──────────┴─────────────────────────────────┘
```

同時維護一個**巨集名稱表（MNT，Macro Name Table）**，將巨集名稱映射到 MDT 中的位置：

```
巨集名稱表（MNT）
┌──────────────┬──────────┬──────────────────┐
│  巨集名稱    │ 引數個數 │  MDT 中的起始位置 │
├──────────────┼──────────┼──────────────────┤
│  PRINT_STR   │    2     │        0          │
│  NEWLINE     │    0     │        5          │
│  CHECK_ZERO  │    1     │       10          │
└──────────────┴──────────┴──────────────────┘
```

### 7.9.2 巨集展開的演算法

```
展開演算法：

對每一行程式碼：
    1. 查找 MNT，判斷這行是否為巨集呼叫
    2. 若是：
        a. 從引數列表提取 %1, %2, ...
        b. 從 MDT 取出巨集主體
        c. 將主體中的 %1, %2... 替換為實際引數
        d. 若主體中有 %%label，產生唯一標號（如 ..@N.label）
        e. 將替換後的文字插入到目前位置（遞迴展開巢狀巨集）
    3. 若否：直接輸出這行（交給組譯器）
```

## 7.10 巨集的常見錯誤與注意事項

### 7.10.1 副作用（Side Effect）問題

```c
// C 巨集的副作用問題
#define MAX(a, b)  ((a) > (b) ? (a) : (b))

int x = 5, y = 3;
int m = MAX(x++, y++);
// 展開：((x++) > (y++) ? (x++) : (y++))
// x++ 可能被求值兩次！→ x 可能從 5 變成 7，而非 6

// 解決：使用 inline 函式（C99/C++）
static inline int max_safe(int a, int b) { return a > b ? a : b; }
int m = max_safe(x++, y++);   // x++ 只求值一次
```

### 7.10.2 組語巨集的暫存器衝突

```asm
; 問題：巨集內部使用的暫存器可能與呼叫者的暫存器衝突
%macro COMPUTE_LEN 1
    MOV  ECX, 0          ; ← 破壞了 ECX！
.loop:
    CMP  BYTE [%1 + ECX], 0
    JE   %%done
    INC  ECX
    JMP  .loop
%%done:
    ; ECX 現在是長度
%endmacro

    MOV  ECX, 10         ; 呼叫者設定 ECX
    COMPUTE_LEN msg      ; ← ECX 被巨集覆蓋！

; 解決：在巨集內 PUSH/POP 所有使用的暫存器
%macro COMPUTE_LEN_SAFE 1
    PUSH RCX             ; 保存呼叫者的 RCX
    PUSH RBX
    ; ... 使用 RBX 代替 RCX 儲存結果 ...
    POP  RBX
    POP  RCX
%endmacro
```

### 7.10.3 巨集遞迴（NASM 不支援）

```asm
; ❌ NASM 的巨集不支援遞迴
%macro RECURSE_BAD 1
    %if %1 > 0
        RECURSE_BAD (%1 - 1)   ; 這會造成無限展開！
    %endif
%endmacro

; ✓ 可以用 %rep 實現有限次重複
%macro UNROLL_LOOP 2   ; count, body_label
    %rep %1
        CALL %2
    %endrep
%endmacro
```

## 7.11 本章小結

**巨集的本質：**
- 巨集是**組譯前的文字替換**，不產生任何執行時開銷
- 巨集處理器維護巨集名稱表（MNT）和巨集定義表（MDT）
- 每次巨集呼叫都展開為一份獨立的程式碼，多次使用程式碼量倍增

**NASM 巨集語法：**
- `%macro 名稱 引數數 ... %endmacro` 定義巨集
- `%1`、`%2`... 引用各引數；`%0` 為引數總數
- `%%label` 產生展開時唯一的標號，避免多次使用時的標號衝突
- `%rep`、`%rotate`、`%assign` 支援進階控制
- `%ifdef`、`%if`、`%error` 支援條件組譯

**C 前置處理器：**
- `#define` 定義物件式或函式式巨集，本質也是文字替換
- 函式式巨集引數必須加括號，防止運算子優先順序問題
- `do { ... } while(0)` 是多行巨集的慣用包裝
- `__VA_ARGS__` 支援可變引數巨集
- `#ifdef`、`#ifndef`、`#pragma once` 用於防止標頭檔重複包含

**巨集 vs 副程式：**
- 短小、熱路徑程式碼 → 巨集（無呼叫開銷）
- 較長、需要遞迴 → 副程式
- 組譯時期決定的行為（條件組譯）→ 巨集唯一選擇
